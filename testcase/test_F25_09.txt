// Constant folding with identity properties
// and constant propagation

main
int a,b,c,d,e;
{
    a = 5;
    b = a * 1; // b = 5
    c = a * 0; // c = 0
    d = a + 0; // d = 5
    e = a ^ 1; // e = 5
    call printInt(a);
    call printInt(b);
    call printInt(c);
    call printInt(d);
    call printInt(e);
}.
